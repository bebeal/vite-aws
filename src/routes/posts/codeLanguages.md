## Zig
```zig
const std = @import("std");

pub fn main() !void {
    std.debug.print("Hello World\n", .{});

    const x: i32 = 5;
    const y: i32 = 10;
    std.debug.print("Sum: {}\n", .{x + y});
}
```

## TypeScript
```typescript
interface Person {
  name: string;
  age: number;
}

function greet(person: Person): string {
  return `Hello, ${person.name}!`;
}

const user: Person = { name: 'World', age: 42 };
console.log(greet(user));
```

## Ruby

    @name = name
  end

  end
